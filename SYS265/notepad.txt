CA Machine:
cd /etc/pki/CA
touch index.txt
echo 1000 > serial

openssl genrsa -des3 -out private/cakey.pem 2048
openssl req -new -x509 -days 365 -key private/cakey.pem -out cacert.pem

Workstation:
cd /tmp
openssl ca -out websrv.crt -infiles websrv.csr
scp websrv.crt psoares@CA_IP:/tmp

CA machine:
cd /tmp
openssl ca -out websrv.crt -infiles websrv.csr
scp websrv.crt psoares@wks_IP:/filepath

Workstation:
websrv.crt -> /etc/pki/tls/certs
websrv.key -> /etc/pki/tls/private
yum -y install mod_ssl
vi /etc/httpd/conf.d/ssl.conf (check path to .crt and .key are correct)
firewall-cmd --add-port=443/tcp --permanent
firewall-cmd --reload
systemctl restart httpd
